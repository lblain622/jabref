 import org.tinylog.core.LogEntry;

 import org.tinylog.writers.AbstractFormatPatternWriter;
 
 public class GuiWriter extends AbstractFormatPatternWriter {
 
     public GuiWriter(final Map<String, String> properties) {
         super(properties);
     }
 
     public GuiWriter() {
         this(Map.of());
     }
 
     @Override

     }

     }

     }

     }
